/**
 * Increment the MAX_PAGES variable.
 * This script is only executed if a "next >>" link was found on the "News Page".
 */
Integer maxPagesInteger = (Integer) session.getVariable("MAX_PAGES");
int maxPages = maxPagesInteger.intValue();

session.log("[INFO] Incrementing MAX_PAGES. New value is: " + (maxPages + 1));

session.setVariable("MAX_PAGES", (maxPages + 1));
