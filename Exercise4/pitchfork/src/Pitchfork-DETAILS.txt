FileWriter out = null;
try {
	session.log("[INFO] Writing DETAILS.");

	String fullText = dataRecord.get("NEWS_TEXT");
	String plainText = fullText.replaceAll("\\<[^>]*>","").replaceAll("\\&.*?\\;", "");

	// Open up the file to be appended to.
	out = new FileWriter(session.getVariable("OUT_PATH"), true);
	out.write ("\t\t\t<article>"+ plainText +"</article>\n");
	out.close();

} catch(Exception e) {
   session.log( "An error occurred while writing the data to a file: " + e.getMessage() );
}
